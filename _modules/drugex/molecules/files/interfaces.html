[docs]    def getGenerator(self):
        with open(self.path, mode="r", encoding="utf-8") as text_file:
            for line in text_file:
                yield line



[docs]    def next(self):
        ret = next(self.generator)
        if self.before:
            ret = self.before(ret)
        ret = self.parseMol(ret)
        if self.after:
            ret = self.after(ret)
        return ret
